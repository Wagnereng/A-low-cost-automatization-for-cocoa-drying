#include <ESP8266WiFi.h>
#include<FirebaseArduino.h>
#include "DHT.h"
#define FIREBASE_HOST "mytcc-43e36.firebaseio.com"                     
#define FIREBASE_AUTH "iPAlTnhnsy3QZ0X6g8hOCcRprRa7TqL9Dp5fXuaV"       
#define WIFI_SSID "AtHome"                                               
#define WIFI_PASSWORD "123456789"                                      
#define DHTPIN1 2 //lembrando que são gpio's e não "digitais(D)"
#define DHTPIN2 0
#define DHTPIN3 4
#define DHTPIN4 5
#define DHTPIN5 16
#define Relay1 10 
int rel1;
#define DHTTYPE DHT22 
DHT dht1(DHTPIN1, DHTTYPE);
DHT dht2(DHTPIN2, DHTTYPE);
DHT dht3(DHTPIN3, DHTTYPE);
DHT dht4(DHTPIN4, DHTTYPE);
DHT dhtex(DHTPIN5, DHTTYPE);

void setup() 
{
  Serial.begin(115200);      
  pinMode(Relay1,OUTPUT);
  digitalWrite(Relay1,HIGH);
  WiFi.begin(WIFI_SSID,WIFI_PASSWORD);
  Serial.print("connecting");
  while (WiFi.status()!=WL_CONNECTED){
    Serial.print(".");
    delay(500);
  }
  dht1.begin();
  dht2.begin();
  dht3.begin();
  dht4.begin();
  dhtex.begin();
  
  Serial.println();
  Serial.print("connected:");
  Serial.println(WiFi.localIP());

  Firebase.begin(FIREBASE_HOST,FIREBASE_AUTH);
  Firebase.setInt("FB1",0);                     
  
}
void firebasereconnect()
{
  Serial.println("Trying to reconnect");
  Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
  }

void loop() 
{
  float h1 = dht1.readHumidity();
  float t1 = dht1.readTemperature();
  float k1 = dht1.readTemperature() + 273.2;
  Firebase.setFloat ("Temp",t1);
  Firebase.setFloat ("Humidity",h1);
  
  float h2 = dht2.readHumidity();
  float t2 = dht2.readTemperature();
  float k2 = dht2.readTemperature() + 273.2;
  Firebase.setFloat ("Temp",t2);
  Firebase.setFloat ("Humidity",h2);
  
  float h3 = dht3.readHumidity();
  float t3 = dht3.readTemperature();
  float k3 = dht3.readTemperature() + 273.2;
  Firebase.setFloat ("Temp",t3);
  Firebase.setFloat ("Humidity",h3);
  float h4 = dht4.readHumidity();
  float t4 = dht4.readTemperature();  
  float k4 = dht4.readTemperature() + 273.2;
  Firebase.setFloat ("Temp",t4);
  Firebase.setFloat ("Humidity",h4);
  
  float hex = dhtex.readHumidity();
  float tex = dhtex.readTemperature();  
  float kex = dhtex.readTemperature() + 273.2;
  Firebase.setFloat ("Temp",tex);
  Firebase.setFloat ("Humidity",hex);
  
  Serial.print("DATA,TIME");
  Serial.print(",");
  Serial.print(h1);
  Serial.print(",");
  Serial.print(t1);
  Serial.print(",");
  Serial.print(k1);
  Serial.print(",");
  Serial.print(h2);
  Serial.print(",");
  Serial.print(t2);
  Serial.print(",");
  Serial.print(k2);
  Serial.print(",");
  Serial.print(h3);
  Serial.print(",");
  Serial.print(t3);
  Serial.print(",");
  Serial.print(k3);
  Serial.print(",");
  Serial.print(h4);
  Serial.print(",");
  Serial.print(t4);
  Serial.print(",");
  Serial.print(k4);
  Serial.print(",");
  Serial.print(hex);
  Serial.print(",");
  Serial.print(tex);
  Serial.print(",");
  Serial.print(kex);
  Serial.println();
  delay(600000);

  if (Firebase.failed())
      {
      Serial.print("Configuração numérica falhou:");
      Serial.println(Firebase.error());
      firebasereconnect();
      return;
      }

 //Lendo as strings de entrada para comunicação com relé
 
  rel1=Firebase.getString("FB1").toInt();  
  if(rel1==1)            
    {
      digitalWrite(Relay1,LOW);
      Serial.println("Relay 1 ON");
    }
  if(rel1==0)              
    {                                      
      digitalWrite(Relay1,HIGH);
      Serial.println("Relay 1 OFF");
    }
}
